\begin{equation}
        \left| \overrightarrow{v}(15) \right| = \left| \begin{bmatrix}
            20\cos(15)\\ -20\sin(15) \\ -\frac{2}{16}15
        \end{bmatrix} \right|
        =
        \left| \begin{bmatrix}
        -15.1938 \\ -13.0058 \\ -1.8750
        \end{bmatrix} \right|
        =
        20.0877 \text{ cms$^{-1}$ }
\end{equation}
