Length = $L$, Mass = $M$, Time = $T$
\begin{itemize}
    \item $[x] = L$
    \item $[m] = M$
    \item $[\frac{d^2x}{dt^2}] = \frac{L}{T^2}$
    \item $[k] = \frac{M}{T^2}$
\end{itemize}
